29 % Reads CAD STL ASCII files, which most CAD programs can export.
30 % Used to create Matlab patches of CAD 3D data.
31 % Returns a vertex list and face list, for Matlab patch command.
32 %
33 % filename = 'hook.stl'; % Example file.
34 %
35 fid=fopen(filename, 'r'); %Open the file, assumes STL ASCII format.
36 if fid == -1
37  error('File could not be opened, check name or path.')
38 end
39 %
40 % Render files take the form:
41 %
42 %solid BLOCK
43 % color 1.000 1.000 1.000
44 % facet
45 % normal 0.000000e+00 0.000000e+00 -1.000000e+00
46 % normal 0.000000e+00 0.000000e+00 -1.000000e+00
47 % normal 0.000000e+00 0.000000e+00 -1.000000e+00
48 % outer loop
49 % vertex 5.000000e-01 -5.000000e-01 -5.000000e-01
50 % vertex -5.000000e-01 -5.000000e-01 -5.000000e-01
51 % vertex -5.000000e-01 5.000000e-01 -5.000000e-01
52 % endloop
53 % endfacet
54 %
55 % The first line is object name, then comes multiple facet and vertex lines.
56 % A color specifier is next, followed by those faces of that color, until
57 % next color line.
58 %
59 CAD_object_name = sscanf(fgetl(fid), '%*s %s'); %CAD object name, if needed.
60 % %Some STLs have it, some don't.
61 vnum=0; %Vertex number counter.
62 report_num=0; %Report the status as we go.
63 VColor = 0;
64 %
65 while feof(fid) == 0 % test for end of file, if not then do stuff
66  tline = fgetl(fid); % reads a line of data from file.
67  fword = sscanf(tline, '%s '); % make the line a character string
68 % Check for color
69  if strncmpi(fword, 'c',1) == 1; % Checking if a "C"olor line, as "C" is 1st char.
70  VColor = sscanf(tline, '%*s %f %f %f'); % & if a C, get the RGB color data of the face.
71  end % Keep this color, until the next color is used.
72  if strncmpi(fword, 'v',1) == 1; % Checking if a "V"ertex line, as "V" is 1st char.

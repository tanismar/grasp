51 void VisualizationThread::run()
52 {
53  boost::shared_ptr<pcl::visualization::PCLVisualizer> viewer(new pcl::visualization::PCLVisualizer("3D Viewer"));
54  int scale=15;
55  viewer->initCameraParameters();
56  viewer->setBackgroundColor (0, 0, 0);
57 
58  viewer->setPosition(x,y);
59  if (sizex!=0 && sizey!=0)
60  viewer->setSize(sizex,sizey);
61 
62  for (unsigned int i=0; i<data.goodPointsIndexes->size(); i++)
63  {
64  ostringstream temp;
65  temp<<i;
66  string s=temp.str();
67  pcl::PointXYZ point(data.cloud->points.at((*data.goodPointsIndexes)[i]).x, data.cloud->points.at((*data.goodPointsIndexes)[i]).y, data.cloud->points.at((*data.goodPointsIndexes)[i]).z);
68  viewer->addSphere (point, 0.002, 1, 1, 0, "pointScored"+s);
69  }
70 
71  pcl::PointXYZ origin(data.chosenPoint[0],data.chosenPoint[1],data.chosenPoint[2]);
72  pcl::PointXYZ normalOrigin;
73  pcl::PointXYZ normalOriginScaled;
74  if (data.hand=="left")
75  {
76  normalOrigin.x=origin.x+data.chosenOrientation(0,2);
77  normalOrigin.y=origin.y+data.chosenOrientation(1,2);
78  normalOrigin.z=origin.z+data.chosenOrientation(2,2);
79  normalOriginScaled.x=origin.x+(data.chosenOrientation(0,2)/scale);
80  normalOriginScaled.y=origin.y+(data.chosenOrientation(1,2)/scale);
81  normalOriginScaled.z=origin.z+(data.chosenOrientation(2,2)/scale);
82  }
83  else
84  {
85  normalOrigin.x=origin.x-data.chosenOrientation(0,2);
86  normalOrigin.y=origin.y-data.chosenOrientation(1,2);
87  normalOrigin.z=origin.z-data.chosenOrientation(2,2);
88  normalOriginScaled.x=origin.x-(data.chosenOrientation(0,2)/scale);
89  normalOriginScaled.y=origin.y-(data.chosenOrientation(1,2)/scale);
90  normalOriginScaled.z=origin.z-(data.chosenOrientation(2,2)/scale);
91  }
92 
93  viewer->addSphere (origin, 0.002, 1, 0, 0, "pointChosen");
94  viewer->addArrow<pcl::PointXYZ,pcl::PointXYZ>(origin,normalOriginScaled,0,0,1,0,"pz");
95 
96  pcl::PointXYZ xaxis(normalOrigin.x+data.chosenOrientation(0,0),normalOrigin.y+data.chosenOrientation(1,0),normalOrigin.z+data.chosenOrientation(2,0));
97  pcl::PointXYZ xaxisScaled(normalOriginScaled.x+(data.chosenOrientation(0,0)/scale),normalOriginScaled.y+(data.chosenOrientation(1,0)/scale),normalOriginScaled.z+(data.chosenOrientation(2,0)/scale));
98 
99  viewer->addArrow<pcl::PointXYZ,pcl::PointXYZ>(xaxisScaled,normalOriginScaled,1,0,0,0,"px");
100 
101  pcl::PointXYZ yaxis(normalOrigin.x+data.chosenOrientation(0,1),normalOrigin.y+data.chosenOrientation(1,1),normalOrigin.z+data.chosenOrientation(2,1));
102  pcl::PointXYZ yaxisScaled(normalOriginScaled.x+(data.chosenOrientation(0,1)/scale),normalOriginScaled.y+(data.chosenOrientation(1,1)/scale),normalOriginScaled.z+(data.chosenOrientation(2,1)/scale));
103 
104  viewer->addArrow<pcl::PointXYZ,pcl::PointXYZ>(yaxisScaled,normalOriginScaled,0,1,0,0,"py");
105 
106  BoundingBox bb(data.boundingBox->getBoundingBox());
107  bb.drawBoundingBox(viewer);
108 
109  viewer->addPointCloudNormals<pcl::PointXYZRGB,pcl::Normal>(data.cloud,data.normals,10,0.01,"normals");
110 
111  pcl::visualization::PointCloudColorHandlerRGBField<pcl::PointXYZRGB> rgb(data.cloud);
112  viewer->addPointCloud<pcl::PointXYZRGB> (data.cloud, rgb, "rgb");
113  viewer->setPointCloudRenderingProperties (pcl::visualization::PCL_VISUALIZER_POINT_SIZE, 3, "rgb");
114 
115  viewer->resetCamera();
116  viewer->setCameraPosition(-0.0611749,-0.040113,0.00667606,-0.105521,0.0891437,0.990413);
117 
118  mutex.wait();
119  running=true;
120  mutex.post();
121 
122  while (!viewer->wasStopped())
123  {
124  mutex.wait();
125  if (!running)
126  {
127  viewer->close();
128  mutex.post();
129  break;
130  }
131  mutex.post();
132  viewer->spinOnce (100);
133  boost::this_thread::sleep (boost::posix_time::microseconds (100000));
134  }
135 }
